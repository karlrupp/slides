  \begin{block}{$\mathfrak{p}$-Bratu Equation}
    \begin{gather*}
      -\nabla \cdot (\eta \nabla u) - \lambda e^u - f = 0 \\
      \eta(\gamma) = (\epsilon^2 + \gamma)^{\frac{\mathfrak{p}-2}{2}} \qquad \gamma(u) = \frac{1}{2} \vert\nabla u\vert^2
    \end{gather*}
