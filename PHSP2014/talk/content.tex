#include <iostream>
#include "viennacl/vector.hpp"

typedef float   NumericT;
using namespace viennacl;

int main()
{
  std::size_t N = 128*1024;
  vector<NumericT> x = scalar_vector<NumericT>(N, 1.0);
  vector<NumericT> y = scalar_vector<NumericT>(N, 2.0);

  x += y;

  std::cout << x << std::endl;
}
